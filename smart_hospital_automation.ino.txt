#include <SoftwareSerial.h>

#define SIM800_TX_PIN 2
#define SIM800_RX_PIN 3

SoftwareSerial serialSIM800(SIM800_TX_PIN, SIM800_RX_PIN);

int sensorPin1 = A0;
int sensorPin2 = A1;
int sensorPin3 = A2;

float tem, hr, bp;

int bzr = 10;
int l1 = 4;
int l2 = 5;
int l3 = 6;

void setup() {

  Serial.begin(9600);

  serialSIM800.begin(9600);
  delay(1000);

  Serial.println("Setup Complete!");
  Serial.println("Sending SMS...");

  serialSIM800.write("AT+CMGF=1\r\n");
  delay(1000);

  serialSIM800.write("AT+CMGS=\"6379995431\"\r\n");
  delay(1000);

  serialSIM800.write("WELCOME TO ACE");
  delay(1000);

  serialSIM800.write((char)26);
  delay(1000);

  Serial.println("SMS Sent!");

  pinMode(bzr, OUTPUT);
  pinMode(l1, OUTPUT);
  pinMode(l2, OUTPUT);
  pinMode(l3, OUTPUT);

  digitalWrite(bzr, LOW);
}

void loop() {

  float temp_val, hr1, bp1;

  hr = analogRead(sensorPin1);
  delay(50);

  bp = analogRead(sensorPin2);
  delay(50);

  tem = analogRead(sensorPin3);
  delay(50);

  temp_val = (tem * 0.4883) / 10;

  hr1 = hr + 70;
  bp1 = bp / 4;

  Serial.println("HEART RATE");
  Serial.println(hr1);

  Serial.println("BP RATE");
  Serial.println(bp1);

  Serial.println("TEMP");
  Serial.println(temp_val);

  delay(500);

  if ((hr1 > 100) && (hr1 < 110)) {

    digitalWrite(bzr, HIGH);
    digitalWrite(l1, HIGH);

  } 
  else if ((bp1 > 120) && (bp1 < 200)) {

    digitalWrite(bzr, HIGH);
    digitalWrite(l2, HIGH);

  } 
  else if (temp_val > 38) {

    digitalWrite(bzr, HIGH);
    digitalWrite(l3, HIGH);

  } 
  else {

    digitalWrite(bzr, LOW);
    digitalWrite(l1, LOW);
    digitalWrite(l2, LOW);
    digitalWrite(l3, LOW);
  }

  delay(500);
}
